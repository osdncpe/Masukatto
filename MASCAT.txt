-- Blobman Beta機能（Kick用）

local BlobmanBeta = {}

function BlobmanBeta.getLocalChar() return LocalPlayer.Character end
function BlobmanBeta.getLocalRoot()
    local char = BlobmanBeta.getLocalChar()
    return char and (char:FindFirstChild("HumanoidRootPart") or char:FindFirstChild("Torso"))
end
function BlobmanBeta.getLocalHum()
    local char = BlobmanBeta.getLocalChar()
    return char and char:FindFirstChildOfClass("Humanoid")
end
function BlobmanBeta.getInv()
    return Workspace:FindFirstChild(LocalPlayer.Name .. "SpawnedInToys")
end

function BlobmanBeta.SetNetworkOwner(part)
    if setNetworkOwnerEvent then
        setNetworkOwnerEvent:FireServer(part, BlobmanBeta.getLocalRoot().CFrame)
    end
end

function BlobmanBeta.ungrab(part)
    if destroyGrabLineEvent then
        destroyGrabLineEvent:FireServer(part)
    end
end

function BlobmanBeta.getBlobman()
    local inv = BlobmanBeta.getInv()
    if inv then
        local v = inv:FindFirstChild("CreatureBlobman")
        if v and v:FindFirstChild("VehicleSeat") then return v end
    end
    for _, p in Workspace.PlotItems:GetChildren() do
        local m = p:FindFirstChild("CreatureBlobman")
        if m and m:FindFirstChild("PlayerValue") and m.PlayerValue.Value == LocalPlayer.Name then
            return m
        end
    end
    return nil
end

function BlobmanBeta.spawnBlobman()
    if SpawnToyRF then
        SpawnToyRF:InvokeServer("CreatureBlobman", BlobmanBeta.getLocalRoot().CFrame, Vector3.zero)
    end
    task.wait(1.0)
    return BlobmanBeta.getBlobman()
end

function BlobmanBeta.destroyBlobman()
    local blob = BlobmanBeta.getBlobman()
    if blob and DeleteToyRE then
        DeleteToyRE:FireServer(blob)
    end
end

function BlobmanBeta.isSittingOnBlobman()
    local hum = BlobmanBeta.getLocalHum()
    if not hum then return false end
    local blob = BlobmanBeta.getBlobman()
    if not blob or not blob:FindFirstChild("VehicleSeat") then return false end
    return hum.Sit and hum.SeatPart == blob.VehicleSeat
end

function BlobmanBeta.ensureSitBlobman()
    local blob = BlobmanBeta.getBlobman()
    if not blob or not blob:FindFirstChild("VehicleSeat") then return false end
    local seat = blob.VehicleSeat
    local hum = BlobmanBeta.getLocalHum()
    if hum and not hum.Sit then
        seat:Sit(hum)
        task.wait(0.6)
    end
    return BlobmanBeta.isSittingOnBlobman()
end

function BlobmanBeta.blobGrab(blob, target, side)
    if not blob then return end
    local detector = blob:FindFirstChild(side .. "Detector")
    if not detector then return end
    local weld = detector:FindFirstChild(side .. "Weld")
    if not weld then return end
    local script = blob:FindFirstChild("BlobmanSeatAndOwnerScript", true)
    if not script then return end
    local remote = script:FindFirstChild("CreatureGrab")
    if remote then
        pcall(function() remote:FireServer(detector, target, weld) end)
    end
end

function BlobmanBeta.blobDrop(blob, target, side)
    if not blob then return end
    local detector = blob:FindFirstChild(side .. "Detector")
    if not detector then return end
    local script = blob:FindFirstChild("BlobmanSeatAndOwnerScript", true)
    if not script then return end
    local remote = script:FindFirstChild("CreatureDrop")
    if remote then
        pcall(function() remote:FireServer(detector, target) end)
    end
end

function BlobmanBeta.blobKick(blob, targetRoot, side)
    BlobmanBeta.blobGrab(blob, BlobmanBeta.getLocalRoot(), side)
    task.wait(0.08)
    BlobmanBeta.SetNetworkOwner(targetRoot)
    task.wait(0.08)
    targetRoot.CFrame = targetRoot.CFrame + Vector3.new(0, 16, 0)
    task.wait(0.08)
    BlobmanBeta.ungrab(targetRoot)
    task.wait(0.08)
    BlobmanBeta.blobGrab(blob, targetRoot, side)
    task.wait(0.08)
    BlobmanBeta.blobDrop(blob, targetRoot, side)
    task.wait(0.08)
    BlobmanBeta.ungrab(targetRoot)
end

local isExecuting = false
local selectedPlayer = nil

local function UpdatePlayerList(Dropdown)
    local playerNames = {}
    for _, plr in Players:GetPlayers() do
        if plr ~= LocalPlayer then
            table.insert(playerNames, plr.Name)
        end
    end
    Dropdown:Refresh(playerNames, true)
end

local function KickSelectedPlayer()
    if isExecuting then
        OrionLib:MakeNotification({Name = "Error", Content = "Already executing...", Time = 3})
        return
    end
    
    if not selectedPlayer then
        OrionLib:MakeNotification({Name = "Error", Content = "Please select a player first", Time = 3})
        return
    end
    
    local targetPlayer = Players:FindFirstChild(selectedPlayer)
    if not targetPlayer then
        OrionLib:MakeNotification({Name = "Error", Content = "Player not found", Time = 3})
        return
    end
    
    local targetChar = targetPlayer.Character
    if not targetChar then
        OrionLib:MakeNotification({Name = "Error", Content = "Target has no character", Time = 3})
        return
    end
    
    local targetRoot = targetChar:FindFirstChild("HumanoidRootPart")
    if not targetRoot then
        OrionLib:MakeNotification({Name = "Error", Content = "Target has no HumanoidRootPart", Time = 3})
        return
    end
    
    isExecuting = true
    
    local function doKick()
        local blob = BlobmanBeta.getBlobman()
        if not blob then
            blob = BlobmanBeta.spawnBlobman()
            if not blob then
                OrionLib:MakeNotification({Name = "Error", Content = "Failed to spawn Blobman", Time = 5})
                return false
            end
        end
        
        if not BlobmanBeta.isSittingOnBlobman() then
            BlobmanBeta.ensureSitBlobman()
            if not BlobmanBeta.isSittingOnBlobman() then
                OrionLib:MakeNotification({Name = "Error", Content = "Not sitting on Blobman", Time = 4})
                return false
            end
        end
        
        local myRoot = BlobmanBeta.getLocalRoot()
        if myRoot and targetRoot then
            local oldPos = myRoot.CFrame
            myRoot.CFrame = targetRoot.CFrame
            task.wait(0.07)
            BlobmanBeta.blobKick(blob, targetRoot, "Left")
            task.wait(0.25)
            myRoot.CFrame = oldPos
        end
        
        BlobmanBeta.destroyBlobman()
        OrionLib:MakeNotification({Name = "Success", Content = "Kicked " .. targetPlayer.Name .. "!", Time = 5})
        return true
    end
    
    doKick()
    isExecuting = false
end

-- Kick All機能（全員キック）
-- ============================================
local currentBlob = nil
local isKickAllActive = false
local playerStatus = {}

local function GetAllPlayers()
    local players = {}
    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= LocalPlayer then
            table.insert(players, player)
        end
    end
    return players
end

local function GetMyRoot()
    return LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
end

local function ResetTargets()
    for id, status in pairs(playerStatus) do
        if status == "Targeting" then playerStatus[id] = nil end
    end
end

local function KickAll()
    if isKickAllActive then return end
    isKickAllActive = true
    
    local allPlayers = GetAllPlayers()
    if #allPlayers == 0 then 
        isKickAllActive = false
        OrionLib:MakeNotification({Name = "Error", Content = "No players to kick", Time = 3})
        return 
    end

    for _, targetPlayer in ipairs(allPlayers) do
        playerStatus[targetPlayer.UserId] = "Targeting"
    end
    
    local rootPart = GetMyRoot()
    if rootPart and SpawnToyRF then
        local spawnPos = rootPart.CFrame * CFrame.new(0, 0, -5)
        SpawnToyRF:InvokeServer("CreatureBlobman", spawnPos, Vector3.new(0, 127, 0))
    end
    task.wait(0.5)
    
    local toyFolder = workspace:FindFirstChild(LocalPlayer.Name .. "SpawnedInToys")
    currentBlob = toyFolder and toyFolder:FindFirstChild("CreatureBlobman")
    if not currentBlob then 
        ResetTargets()
        isKickAllActive = false
        OrionLib:MakeNotification({Name = "Error", Content = "Failed to spawn Blobman", Time = 3})
        return 
    end
    
    local vehicleSeat = currentBlob:FindFirstChild("VehicleSeat")
    if vehicleSeat and LocalPlayer.Character then
        local humanoid = LocalPlayer.Character:FindFirstChildOfClass("Humanoid")
        if humanoid then
            vehicleSeat:Sit(humanoid)
        end
    end
    task.wait(0.3)
    
    local myRoot = GetMyRoot()
    if not myRoot then 
        ResetTargets()
        isKickAllActive = false
        return 
    end
    
    local tpWait = 0.025
    
    for _, targetPlayer in ipairs(allPlayers) do
        local targetRoot = targetPlayer.Character and targetPlayer.Character:FindFirstChild("HumanoidRootPart")
        if targetRoot then
            myRoot.CFrame = targetRoot.CFrame
            myRoot.AssemblyAngularVelocity = Vector3.zero
            task.wait(tpWait)
            
            local distance = (myRoot.Position - targetRoot.Position).Magnitude
            if distance <= 35 then
                for i = 1, 8 do
                    if not targetRoot.Parent or not currentBlob.Parent then break end
                    pcall(function()
                        local script = currentBlob:FindFirstChild("BlobmanSeatAndOwnerScript", true)
                        if script then
                            local grab = script:FindFirstChild("CreatureGrab")
                            local release = script:FindFirstChild("CreatureRelease")
                            if grab and release then
                                grab:FireServer(currentBlob.LeftDetector, targetRoot, currentBlob.LeftDetector.LeftWeld)
                                release:FireServer(currentBlob.LeftDetector.LeftWeld)
                            end
                        end
                    end)
                    task.wait()
                end
            end
        end
    end
    
    myRoot.CFrame = CFrame.new(0, 100, 0)
    myRoot.AssemblyLinearVelocity = Vector3.zero
    task.wait(0.1)
    
    for _, part in ipairs(currentBlob:GetDescendants()) do
        if part:IsA("BasePart") then pcall(function() part.Anchored = true end) end
    end
    task.wait(0.1)
    
    local radius = 15
    for i, targetPlayer in ipairs(allPlayers) do
        local targetRoot = targetPlayer.Character and targetPlayer.Character:FindFirstChild("HumanoidRootPart")
        if targetRoot then
            local angle = math.rad((i - 1) * (360 / #allPlayers))
            local x = radius * math.cos(angle)
            local z = radius * math.sin(angle)
            targetRoot.CFrame = CFrame.new(x, 110, z)
        end
    end
    task.wait(0.1)
    
    for _ = 1, 2 do
        for _, targetPlayer in ipairs(allPlayers) do
            local targetRoot = targetPlayer.Character and targetPlayer.Character:FindFirstChild("HumanoidRootPart")
            if targetRoot then
                task.spawn(function()
                    pcall(function()
                        if setNetworkOwnerEvent then
                            setNetworkOwnerEvent:FireServer(targetRoot, CFrame.new(targetRoot.Position))
                        end
                        if destroyGrabLineEvent then
                            destroyGrabLineEvent:FireServer(targetRoot)
                        end
                    end)
                end)
            end
        end
        task.wait(0.1)
    end
    
    task.wait(0.3)
    
    for _, targetPlayer in ipairs(allPlayers) do
        local targetRoot = targetPlayer.Character and targetPlayer.Character:FindFirstChild("HumanoidRootPart")
        if targetRoot then
            task.spawn(function()
                pcall(function()
                    local script = currentBlob:FindFirstChild("BlobmanSeatAndOwnerScript", true)
                    if script then
                        local grab = script:FindFirstChild("CreatureGrab")
                        if grab then
                            grab:FireServer(currentBlob.LeftDetector, targetRoot, currentBlob.LeftDetector.LeftWeld)
                            grab:FireServer(currentBlob.RightDetector, targetRoot, currentBlob.RightDetector.RightWeld)
                        end
                    end
                end)
            end)
        end
    end
    
    task.wait(0.1)

    for _, targetPlayer in ipairs(allPlayers) do
        if targetPlayer and targetPlayer.Parent == Players then
            playerStatus[targetPlayer.UserId] = "Kicked"
        end
    end
    
    for _, part in ipairs(currentBlob:GetDescendants()) do
        if part:IsA("BasePart") then pcall(function() part.Anchored = false end) end
    end
    
    task.wait(1)
    isKickAllActive = false
    OrionLib:MakeNotification({Name = "Success", Content = "Kicked all players!", Time = 5})
end

-- Spam Kick（スパムキック）機能
-- ============================================
local spamTarget = nil
local currentSpamTask = nil
local currentMainTask = nil
local currentBodyPosition = nil
local currentBodyGyro = nil
local savedPosition = nil
local dragging = false
local grabStartTime = 0
local checkStartTime = 0
local isSpamActive = false
local noclipConnection = nil

local function SetNoclip(enabled)
    if noclipConnection then
        noclipConnection:Disconnect()
        noclipConnection = nil
    end
    
    if enabled then
        noclipConnection = RunService.Stepped:Connect(function()
            if LocalPlayer.Character then
                for _, part in ipairs(LocalPlayer.Character:GetDescendants()) do
                    if part:IsA("BasePart") then
                        part.CanCollide = false
                    end
                end
            end
        end)
    end
end

local function SetNetworkOwnerForTarget(targetRoot)
    if not targetRoot then return end
    if not setNetworkOwnerEvent then return end
    pcall(function()
        setNetworkOwnerEvent:FireServer(targetRoot, targetRoot.CFrame)
    end)
end

local function DoDestroyGrabLine(targetRoot)
    if not targetRoot then return end
    if not destroyGrabLineEvent then return end
    pcall(function()
        destroyGrabLineEvent:FireServer(targetRoot)
    end)
end

local function CreateBodies(targetRoot, pos)
    if currentBodyPosition then
        currentBodyPosition:Destroy()
        currentBodyPosition = nil
    end
    if currentBodyGyro then
        currentBodyGyro:Destroy()
        currentBodyGyro = nil
    end
    
    currentBodyPosition = Instance.new("BodyPosition")
    currentBodyPosition.MaxForce = Vector3.new(1e8, 1e8, 1e8)
    currentBodyPosition.P = 100000
    currentBodyPosition.D = 1000
    currentBodyPosition.Position = pos
    currentBodyPosition.Parent = targetRoot
    
    currentBodyGyro = Instance.new("BodyGyro")
    currentBodyGyro.MaxTorque = Vector3.new(1e8, 1e8, 1e8)
    currentBodyGyro.P = 100000
    currentBodyGyro.D = 1000
    currentBodyGyro.CFrame = CFrame.new(pos)
    currentBodyGyro.Parent = targetRoot
end

local function CleanupBodies()
    if currentBodyPosition then
        currentBodyPosition:Destroy()
        currentBodyPosition = nil
    end
    if currentBodyGyro then
        currentBodyGyro:Destroy()
        currentBodyGyro = nil
    end
end

local function StopSpamAll()
    isSpamActive = false
    if currentSpamTask then
        task.cancel(currentSpamTask)
        currentSpamTask = nil
    end
    if currentMainTask then
        task.cancel(currentMainTask)
        currentMainTask = nil
    end
    CleanupBodies()
    SetNoclip(false)
    dragging = false
    grabStartTime = 0
    checkStartTime = 0
end

local function RunSpamKick()
    if isSpamActive then return end
    
    if not spamTarget then
        OrionLib:MakeNotification({Name = "Error", Content = "Please select a player first", Time = 3})
        return
    end
    
    isSpamActive = true
    SetNoclip(true)
    
    currentMainTask = task.spawn(function()
        local target = spamTarget
        if not target then 
            StopSpamAll()
            return 
        end
        
        local myChar = LocalPlayer.Character
        local myRoot = myChar and myChar:FindFirstChild("HumanoidRootPart")
        if not myRoot then 
            StopSpamAll()
            return 
        end
        
        savedPosition = myRoot.CFrame
        dragging = false
        grabStartTime = 0
        checkStartTime = 0
        
        local spamKickRunning = false
        local currentTargetRoot = nil
        
        while isSpamActive do
            local currentTargetPlayer = Players:FindFirstChild(target.Name)
            if not currentTargetPlayer then
                CleanupBodies()
                spamKickRunning = false
                task.wait()
                continue
            end
            
            myChar = LocalPlayer.Character
            myRoot = myChar and myChar:FindFirstChild("HumanoidRootPart")
            local tChar = currentTargetPlayer.Character
            local tRoot = tChar and tChar:FindFirstChild("HumanoidRootPart")
            local tHum = tChar and tChar:FindFirstChild("Humanoid")
            
            if tRoot and tHum and tHum.Health > 0 and myRoot then
                if not spamKickRunning then
                    spamKickRunning = true
                    currentTargetRoot = tRoot
                    currentSpamTask = task.spawn(function()
                        while isSpamActive and tRoot and tRoot.Parent do
                            if tRoot and tRoot.Parent then
                                DoDestroyGrabLine(tRoot)
                            end
                            task.wait()
                            
                            if tRoot and tRoot.Parent then
                                SetNetworkOwnerForTarget(tRoot)
                            end
                            task.wait()
                        end
                    end)
                end
                
                if not dragging then
                    myRoot.CFrame = tRoot.CFrame * CFrame.new(0, 0, 3)
                    CleanupBodies()
                    checkStartTime = 0
                    
                    pcall(function()
                        SetNetworkOwnerForTarget(tRoot)
                        SetNetworkOwnerForTarget(tRoot)
                        DoDestroyGrabLine(tRoot)
                    end)
                    
                    myRoot.AssemblyLinearVelocity = Vector3.zero
                    myRoot.AssemblyAngularVelocity = Vector3.zero
                    
                    if grabStartTime == 0 then grabStartTime = tick() end
                    if tick() - grabStartTime > 0.35 then
                        dragging = true
                        grabStartTime = 0
                        checkStartTime = tick()
                        local lockPos = savedPosition * CFrame.new(0, 17, 0)
                        CreateBodies(tRoot, lockPos.Position)
                    end
                else
                    myRoot.CFrame = savedPosition
                    local lockPos = savedPosition * CFrame.new(0, 17, 0)
                    
                    myRoot.AssemblyLinearVelocity = Vector3.zero
                    myRoot.AssemblyAngularVelocity = Vector3.zero
                    
                    if currentBodyPosition and currentBodyPosition.Parent then
                        currentBodyPosition.Position = lockPos.Position
                        if currentBodyGyro then
                            currentBodyGyro.CFrame = lockPos
                        end
                    else
                        CreateBodies(tRoot, lockPos.Position)
                    end
                    
                    if checkStartTime > 0 and tick() - checkStartTime > 0.30 then
                        local currentDist = (tRoot.Position - lockPos.Position).Magnitude
                        if currentDist > 10 then
                            dragging = false
                            grabStartTime = 0
                            checkStartTime = 0
                            CleanupBodies()
                            myRoot.CFrame = tRoot.CFrame * CFrame.new(0, 0, 3)
                            currentTargetRoot = tRoot
                        else
                            checkStartTime = tick()
                        end
                    end
                end
            else
                dragging = false
                grabStartTime = 0
                checkStartTime = 0
                CleanupBodies()
                spamKickRunning = false
                currentTargetRoot = nil
            end
            
            task.wait()
        end
        
        CleanupBodies()
        SetNoclip(false)
    end)
end

local function StopSpamKick()
    if isSpamActive then
        StopSpamAll()
        OrionLib:MakeNotification({Name = "Spam Kick", Content = "Stopped", Time = 2})
    end
end

local function setMassless()
    local char = LocalPlayer.Character
    if not char then return end
    
    for _, prt in pairs(char:GetChildren()) do
        if prt:IsA("BasePart") and prt.Massless then
            prt.Massless = false
            dropFromBlobs()
        end
    end
end

local function moveRootAttachment()
    local char = LocalPlayer.Character
    if not char then return end
    
    local hrp = char:FindFirstChild("HumanoidRootPart")
    local truePart = char:FindFirstChild("TruePositionPart")
    
    if hrp and truePart then
        local rootAttachment = hrp:FindFirstChild("RootAttachment")
        
        if rootAttachment then
            task.wait(0.2)
            rootAttachment.Parent = truePart
        end
    end
end

local function restoreRootAttachment()
    local char = LocalPlayer.Character
    if not char then return end
    
    local hrp = char:FindFirstChild("HumanoidRootPart")
    local truePart = char:FindFirstChild("TruePositionPart")
    
    if hrp and truePart then
        local rootAttachment = truePart:FindFirstChild("RootAttachment")
        
        if rootAttachment then
            rootAttachment.Parent = hrp
        end
        
        truePart:Destroy()
    end
end

local function antiblobLoop()
    while antiblob do
        pcall(function()
            if LocalPlayer.Character then
                createTruePospart()
                setMassless()
                
                local hrp = LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
                if hrp and hrp:FindFirstChild("RootAttachment") then
                    local truePart = LocalPlayer.Character:FindFirstChild("TruePositionPart")
                    if truePart and not truePart:FindFirstChild("RootAttachment") then
                        moveRootAttachment()
                    end
                end
            end
        end)
        
        task.wait(0.1)
    end
end

local function toggleAntiBlob(enabled)
    antiblob = enabled
    
    if enabled then
        if antiblobConnection then
            antiblobConnection:Disconnect()
        end
        
        task.spawn(antiblobLoop)
        
        antiblobConnection = LocalPlayer.CharacterAdded:Connect(function(char)
            if antiblob then
                task.wait(1)
                createTruePospart()
                task.wait(0.5)
                moveRootAttachment()
            end
        end)
        
        OrionLib:MakeNotification({Name = "Anti-Blob", Content = "Enabled", Time = 3})
    else
        if antiblobConnection then
            antiblobConnection:Disconnect()
            antiblobConnection = nil
        end
        
        restoreRootAttachment()
        OrionLib:MakeNotification({Name = "Anti-Blob", Content = "Disabled", Time = 3})
    end
end

local antiGrabEnabled = false
local antiGrabCharacterConnection = nil

local function setupAntiGrab(char)
    if not char then return end
    
    local function FWC(Parent, Name, Time)
        return Parent:FindFirstChild(Name) or Parent:WaitForChild(Name, Time or 5)
    end
    
    local hrp  = FWC(char, "HumanoidRootPart")
    local hum  = FWC(char, "Humanoid")
    local head = FWC(char, "Head")
    
    if not (hrp and hum and head) then return end
    
    if not struggleEvent or not ragdollRemoteEvent then return end
    
    local AntiGrabProc = false
    local AGWalk = false
    
    head.ChildAdded:Connect(function(PartOwner)
        if PartOwner.Name == "PartOwner" then
            if not AntiGrabProc then
                AntiGrabProc = true
                hum.Sit = false
                struggleEvent:FireServer(LocalPlayer)
                task.spawn(function()
                    while (head and head:FindFirstChild("PartOwner")) or 
                          (isHeldValue and isHeldValue.Value) do
                        struggleEvent:FireServer(LocalPlayer)
                        ragdollRemoteEvent:FireServer(hrp, 0)
                        task.wait()
                    end
                end)
                hrp.Anchored = true
                if not AGWalk then
                    AGWalk = true
                    while (isHeldValue and isHeldValue.Value) and task.wait() do
                        hrp.CFrame = hrp.CFrame + hum.MoveDirection * 0.43
                    end
                end
                hrp.Anchored = false
                AntiGrabProc = false
                AGWalk = false
            end
        end
    end)
    
    local ragdolled = hum:FindFirstChild("Ragdolled")
    if ragdolled then
        ragdolled.Changed:Connect(function()
            if ragdolled.Value then
                for _, v in pairs(char:GetChildren()) do
                    if v:IsA("BasePart") and v:FindFirstChild("BallSocketConstraint") and v.Name ~= "Head" then
                        local constraint = v:FindFirstChild("BallSocketConstraint")
                        if constraint then
                            constraint.Enabled = false
                        end
                        if v:FindFirstChild("RagdollLimbPart") then
                            local weldConstraint = v.RagdollLimbPart:FindFirstChild("WeldConstraint")
                            if weldConstraint then
                                weldConstraint.Enabled = false
                            end
                        end
                    end
                end
            end
        end)
    end
    
    local weldHRP = hrp:FindFirstChild("WeldHRP")
    if weldHRP then
        weldHRP.Changed:Connect(function()
            if weldHRP.Enabled then
                while not hum.Sit do task.wait() end
                hum.Sit = false
                hum.AutoRotate = true
                hum.HipHeight = 1
                while weldHRP.Enabled and task.wait() do
                    head.CFrame = hrp.CFrame + Vector3.new(0, 1.35, 0)
                end
                hum.HipHeight = 0
            end
        end)
    end
    
    for _, v in pairs(char:GetChildren()) do
        if v:IsA("BasePart") and v:FindFirstChild("BallSocketConstraint") and v.Name ~= "Head" then
            local constraint = v:FindFirstChild("BallSocketConstraint")
            if constraint then
                constraint.Enabled = false
            end
            if v:FindFirstChild("RagdollLimbPart") then
                local weldConstraint = v.RagdollLimbPart:FindFirstChild("WeldConstraint")
                if weldConstraint then
                    weldConstraint.Enabled = false
                end
            end
        end
    end
end

local function toggleAntiGrab(enabled)
    antiGrabEnabled = enabled
    
    if enabled then

            antiGrabCharacterConnection:Disconnect()

            antiGrabCharacterConnection:Disconnect()
            antiGrabCharacterConnection = nil
        end
        
        OrionLib:MakeNotification({Name = "Anti-Grab", Content = "Disabled", Time = 3})
    end
end

-- キルバイパス（Kill Bypass）機能 - 高性能版
-- ============================================
local killBypassActive = false
local killBypassPlatformPart = nil
local killBypassRemoteBlockList = {}
local killBypassProtectionEnabled = false
local killBypassLoopInterval = 0.016

-- キャラクター保護（Humanoid監視）
local function protectHumanoid(char)
    local hum = char:FindFirstChild("Humanoid")
    if not hum then return end
    
    local lastHealth = hum.Health
    local lastMaxHealth = hum.MaxHealth
    local isProtected = true
    
    hum:GetPropertyChangedSignal("Health"):Connect(function()
        if not isProtected then return end
        if hum.Health < lastHealth and hum.Health > 0 and hum.Health < lastMaxHealth * 0.9 then
            hum.Health = math.max(hum.Health, lastHealth - 5)
            hum.Health = lastHealth
        end
        lastHealth = hum.Health
    end)
    
    local oldBreak = hum.BreakJoints
    hum.BreakJoints = function()
        if isProtected and killBypassActive then
            return
        end
        return oldBreak(hum)
    end
    
    return function()
        isProtected = false
    end
end

-- リモートイベント遮断
local function blockKillRemotes()
    local killKeywords = {"kill", "damage", "death", "loop", "grab", "explosion", "void", "banish", "delete"}
    
    for _, v in pairs(game:GetDescendants()) do
        if (v:IsA("RemoteEvent") or v:IsA("RemoteFunction")) and not killBypassRemoteBlockList[v] then
            local name = v.Name:lower()
            local isKillRemote = false
            
            for _, keyword in pairs(killKeywords) do
                if name:find(keyword) then
                    isKillRemote = true
                    break
                end
            end
            
            if isKillRemote then
                killBypassRemoteBlockList[v] = true
                
                if v:IsA("RemoteEvent") then
                    v.OnServerEvent = function() return end
                    v.InvokeServer = function() return end
                elseif v:IsA("RemoteFunction") then
                    v.OnServerInvoke = function() return end
                end
            end
        end
    end
end

-- リモートスキャンを定期実行
local function startRemoteScanner()
    spawn(function()
        while true do
            task.wait(15)
            if killBypassActive then
                blockKillRemotes()
            end
        end
    end)
end

-- メインループ
local function StartKillBypassLoop()
    while killBypassActive and LocalPlayer and LocalPlayer.Character do
        local char = LocalPlayer.Character
        local root = char and char:FindFirstChild("HumanoidRootPart")
        local camera = workspace.CurrentCamera
        
        if root and camera and root.Position.Y < 1000000 then
            if not killBypassPlatformPart or not killBypassPlatformPart.Parent then
                killBypassPlatformPart = Instance.new("Part", workspace)
                killBypassPlatformPart.Name = "SkyBase"
                killBypassPlatformPart.Anchored = true
                killBypassPlatformPart.CanCollide = false
                killBypassPlatformPart.Transparency = 1
                killBypassPlatformPart.Size = Vector3.new(1500, 2, 1500)
                killBypassPlatformPart.CFrame = CFrame.new(0, 1000000, 0)
                workspace.FallenPartsDestroyHeight = -9999999
            end

            local originalPos = root.CFrame
            local originalVelocity = root.AssemblyLinearVelocity
            local originalRotVelocity = root.AssemblyAngularVelocity
            
            local savedCameraCFrame = camera.CFrame
            camera.CameraType = Enum.CameraType.Scriptable
            camera.CFrame = savedCameraCFrame
            
            root.CFrame = killBypassPlatformPart.CFrame + Vector3.new(0, 5, 0)
            root.AssemblyLinearVelocity = Vector3.zero
            root.AssemblyAngularVelocity = Vector3.zero
            
            task.wait(killBypassLoopInterval)
            
            if not killBypassActive then break end
            
            char = LocalPlayer.Character
            root = char and char:FindFirstChild("HumanoidRootPart")
            if root then
                root.CFrame = originalPos
                root.AssemblyLinearVelocity = originalVelocity or Vector3.zero
                root.AssemblyAngularVelocity = originalRotVelocity or Vector3.zero
            end
            
            camera.CameraType = Enum.CameraType.Custom
            
            task.wait(killBypassLoopInterval)
        else
            task.wait(0.1)
        end
    end
    
    if killBypassPlatformPart then
        killBypassPlatformPart:Destroy()
        killBypassPlatformPart = nil
    end
    if workspace.CurrentCamera then
        workspace.CurrentCamera.CameraType = Enum.CameraType.Custom
    end
end

-- 全保護機能を有効化
local function enableAllProtections()
    if killBypassProtectionEnabled then return end
    killBypassProtectionEnabled = true
    
    if LocalPlayer.Character then
        protectHumanoid(LocalPlayer.Character)
    end
    
    LocalPlayer.CharacterAdded:Connect(function(char)
        task.wait(0.5)
        protectHumanoid(char)
        task.wait(0.5)
    end)
    
    task.wait(0.5)
    blockKillRemotes()
    startRemoteScanner()
end

local function toggleKillBypass(enabled)
    killBypassActive = enabled
    
    if enabled then
        enableAllProtections()
        task.spawn(StartKillBypassLoop)
        OrionLib:MakeNotification({Name = "Kill Bypass", Content = "Enabled (High Performance)", Time = 3})
    else
        killBypassProtectionEnabled = false
        OrionLib:MakeNotification({Name = "Kill Bypass", Content = "Disabled", Time = 3})
    end
end

    end
    StableGucciSettings.sitConnection = RunService.Heartbeat:Connect(function()

local function CleanUpOldToys()
    if not DeleteToyRE then return end
    local folder = Workspace:FindFirstChild(LocalPlayer.Name .. "SpawnedInToys")
    if not folder then return end
    for _, toy in pairs(folder:GetChildren()) do
        if toy.Name == "CreatureBlobman" then
            pcall(function() 
                DeleteToyRE:FireServer(toy) 
            end)
        end
    end
end

local function RunStableGucci()
    if StableGucciSettings.systemOn then return end
    local char = LocalPlayer.Character or LocalPlayer.CharacterAdded:Wait()
    local hum = char:FindFirstChild("Humanoid")
    local root = char:FindFirstChild("HumanoidRootPart")
    if not hum then hum = char:WaitForChild("Humanoid", 3) end
    if not root then root = char:WaitForChild("HumanoidRootPart", 3) end
    if not (hum and root) then 
        StableGucciSettings.systemOn = false 
        return 
    end
    
    StableGucciSettings.systemOn = true
    
    OrionLib:MakeNotification({
        Name = "Stable Gucci",
        Content = "起動中...",
        Time = 3
    })
    
    CleanUpOldToys()
    
    local savedCFrame = root.CFrame
    local spawnCFrame = CFrame.new(103.85, -7.45, -538.58)
    
    task.spawn(function()
        if SpawnToyRF then
            pcall(function()
                SpawnToyRF:InvokeServer("CreatureBlobman", spawnCFrame, Vector3.zero)
            end)
        end
    end)
    
    local blobman = nil
    local folder = Workspace:FindFirstChild(LocalPlayer.Name .. "SpawnedInToys")
    if not folder then
        folder = Workspace:WaitForChild(LocalPlayer.Name .. "SpawnedInToys", 5)
    end
    if not folder then 
        StableGucciSettings.systemOn = false 
        OrionLib:MakeNotification({
            Name = "Stable Gucci",
            Content = "フォルダが見つかりませんでした",
            Time = 3
        })
        return 
    end
    
    local startTick = tick()
    while not blobman and tick() - startTick < 3 do
        local target = folder:FindFirstChild("CreatureBlobman")
        if target then blobman = target break end
        RunService.Heartbeat:Wait()
    end
    
    if not blobman then 
        StableGucciSettings.systemOn = false
        OrionLib:MakeNotification({
            Name = "Stable Gucci",
            Content = "Blobmanが見つかりませんでした",
            Time = 3
        })
        return 
    end
    
    local head = blobman:FindFirstChild("Head")
    if head then head.Anchored = true end
    
    local seat = blobman:FindFirstChild("VehicleSeat")
    if not seat then seat = blobman:WaitForChild("VehicleSeat", 2) end
    if not seat then 
        StableGucciSettings.systemOn = false 
        OrionLib:MakeNotification({
            Name = "Stable Gucci",
            Content = "シートが見つかりませんでした",
            Time = 3
        })
        return 
    end
    StableGucciSettings.currentSeat = seat
    
    StartRagdollSpam(root)
    root.CFrame = seat.CFrame + Vector3.new(0, 2, 0)
    seat:Sit(hum)
    StartSitSpam(hum, seat)
    
    task.wait(0.6)
    hum.Sit = false
    hum.PlatformStand = true
    
    if seat then 
        seat.Disabled = true 
        pcall(function() seat:Destroy() end)
    end
    if blobman then
        for _, v in pairs(blobman:GetDescendants()) do
            if v:IsA("Weld") or v:IsA("Snap") then 
                v:Destroy() 
            end
        end
    end
    
    root.CFrame = savedCFrame
    root.Velocity = Vector3.zero
    hum.PlatformStand = false
    
    task.wait(0.5)
    if StableGucciSettings.sitConnection then 
        StableGucciSettings.sitConnection:Disconnect() 
        StableGucciSettings.sitConnection = nil 
    end
    if StableGucciSettings.ragdollConnection then 
        StableGucciSettings.ragdollConnection:Disconnect() 
        StableGucciSettings.ragdollConnection = nil 
    end
    
    task.wait(0.5)
    StableGucciSettings.systemOn = false
    
    OrionLib:MakeNotification({
        Name = "Stable Gucci",
        Content = "完了しました！",
        Time = 3
    })
end
